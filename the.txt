#!/bin/perl
use strict;
use warnings;

my $now = localtime;
print "The date is $now\n";

sub calculator {
    my ($operator, @numbers) = @_;
    
    if ($operator eq '+') {
        return add(@numbers);
    }
    elsif ($operator eq '*') {
        return multiply(@numbers);
    }
    else {
        die "Invalid operator: $operator. Use '+' or '*'.\n";
    }
}

sub add {
    my @numbers = @_;
    my $sum = 0;
    $sum += $_ for @numbers;
    return $sum;
}

sub multiply {
    my @numbers = @_;
    my $product = 1;
    $product *= $_ for @numbers;
    return $product;
}

print "Sum of 1, 2, 3, 4: ", calculator('+', 1, 2, 3, 4), "\n";
print "Product of 1, 2, 3, 4: ", calculator('*', 1, 2, 3, 4), "\n";
