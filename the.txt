#!/bin/perl
use strict;
use warnings;

my $now = localtime;
print "The date is $now\n\n";

if (@ARGV != 2) {
    die "Usage: $0 <number1> <number2>\n";
}

my ($num1, $num2) = @ARGV;

if ($num1 !~ /^-?\d+$/ || $num2 !~ /^-?\d+$/) {
    die "Error: Both arguments must be whole numbers\n";
}

my $sum = $num1 + $num2;
my $product = $num1 * $num2;

my $percent;
if ($num2 == 0) {
    $percent = "N/A (div by 0)";
} else {
    $percent = sprintf("%.2f%%", ($num1 / $num2) * 100);
}

printf "%-10s %-10s %-10s %-10s %-15s\n", "Number 1", "Number 2", "Sum", "Product", "Percent";
printf "%-10s %-10s %-10s %-10s %-15s\n", "-" x 10, "-" x 10, "-" x 10, "-" x 10, "-" x 15;

printf "%-10d %-10d %-10d %-10d %-15s\n", $num1, $num2, $sum, $product, $percent;
