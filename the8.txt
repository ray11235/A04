#!/bin/perl

use strict;
use warnings;

my $now = localtime;
print "The date is $now\n\n";

printf "%-8s %-10s %-10s\n", "Number", "Square", "Cube";
print "-" x 28 . "\n";

for (my $i = 4; $i <= 100; $i += 4) {
    my $square = $i * $i;
    my $cube   = $i * $i * $i;
    printf "%-8d %-10d %-10d\n", $i, $square, $cube;
}
