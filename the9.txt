#!/bin/perl

use strict;
use warnings;

my $now = localtime;
print "The current date and time is: $now\n\n";

my @first_names = qw(Tom Will Lady Brad Leonardo Julia Scarlett Natalie Emma Ryan);
my @last_names  = qw(Cruise Smith Gaga Pitt DiCaprio Roberts Johansson Portman Watson Gosling);

my $index = 1;
while (my ($i, $first) = each @first_names) {
    my $last = $last_names[$i];   # same index in the other array
    print "Person #$index: $first $last\n";
    $index++;
}
